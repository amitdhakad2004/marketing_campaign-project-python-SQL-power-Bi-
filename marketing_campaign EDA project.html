def categorize_roi(ROI):
    if ROI >= 650:
        return 'High ROI'
    elif ROI >= 350:
        return 'Moderate ROI'
    else:
        return 'Low ROI'

df['ROI_Category'] = df['ROI'].apply(categorize_roi)
ax = sns.countplot(data=df, x='ROI_Category' , color = 'orange')
